Para ello, se integran los mdulos de Caractersticas de la vivienda y hogar (Mdulo 1) y Gasto en alimentos y bebidas (Mdulo 7).

import numpy as np
import scipy.stats as stats

# Filtrar los datos para la categora "Casa independiente"
casa_independiente = data_combinada[data_combinada['Tipo_Vivienda'] == 'Casa independiente']['log_gasto_bebidas']

# Calcular el promedio general
promedio_general = data_combinada['log_gasto_bebidas'].mean()

# Estadsticas descriptivas de "Casa independiente"
media_casa = casa_independiente.mean()
desv_est_casa = casa_independiente.std()
n_casa = len(casa_independiente)

print("Estadsticas de 'Casa independiente':")
print(f"Media: {media_casa:.2f}, Desviacin Estndar: {desv_est_casa:.2f}, N: {n_casa}")

# Prueba de hiptesis (t-test para una muestra)
t_stat, p_value = stats.ttest_1samp(casa_independiente, promedio_general)
print("\nPrueba de Hiptesis:")
print(f"Estadstico t: {t_stat:.3f}, Valor-p: {p_value:.3f}")

# Interpretacin de la prueba
alpha = 0.05
if p_value < alpha:
    print(f"\nRechazamos H: La media del gasto en 'Casa independiente' es significativamente diferente del promedio general (={alpha}).")
else:
    print(f"\nNo se puede rechazar H: No hay evidencia suficiente para afirmar que la media del gasto en 'Casa independiente' es diferente del promedio general (={alpha}).")

# Calcular intervalo de confianza del 95% para "Casa independiente"
confianza = 0.95
error_estandar = desv_est_casa / np.sqrt(n_casa)
ic_min, ic_max = stats.t.interval(confianza, df=n_casa-1, loc=media_casa, scale=error_estandar)

print("\nIntervalo de Confianza del 95% para la media de 'Casa independiente':")
print(f"[{ic_min:.2f}, {ic_max:.2f}]")
